# Full File Structure and Explanations

`colors` -> This is where Saved Colors goes to
    `icon` -> This is where the Saved Colors for the
        Live Icon Editor's Live Colors Pal
    `json` -> This is where Saved Colors go to when
        you use the black Saved Colors boxes
`custom_pals` -> This is where your fashion+hair is
     saved
    `fashion` -> Where fashion palettes are saved
        and loaded
    `hair` -> Where Hair palettes are saved/loaded
`full_pal` -> This is where your Export Palette button
    goes to
`icons` -> Where your icon BMPs go to
`images` -> Where your full character export previews
    are saved to
